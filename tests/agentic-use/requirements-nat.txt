# NAT (NeMo Agent Toolkit) dependencies for agentic-use eval tasks.
#
# WHY THIS FILE EXISTS INSTEAD OF pyproject.toml
# -----------------------------------------------
# Keep nvidia-nat isolated from the main workspace lock so agentic-use eval
# tasks can iterate without perturbing platform runtime dependency resolution.
#
# Instead, nvidia-nat is installed:
#   1. Inside the Docker container via Dockerfile.agentic-base (runtime)
#   2. Via this file for local development / CI outside the container
#
# INSTALLATION (outside container)
# ---------------------------------
# Install into a separate virtual environment:
#
#   python3.11 -m venv .venv-nat
#   source .venv-nat/bin/activate
#   pip install -r tests/agentic-use/requirements-nat.txt
#   nat --version
#
# OR use uv with an isolated venv:
#
#   uv venv .venv-nat --python 3.11
#   uv pip install -r tests/agentic-use/requirements-nat.txt --python .venv-nat/bin/python
#
# DOCKER CONTAINER
# ----------------
# Dockerfile.agentic-base installs this automatically during image build:
#   uv pip install "nvidia-nat[mcp,eval,langchain]"

nvidia-nat[mcp,eval,langchain]
nvidia-nat-atif
